  \vfill
  \centering
  \begin{beamercolorbox}[sep=8pt,center,shadow=true,rounded=true]{title}
    \usebeamerfont{title}{\huge \color{red} \insertsectionhead} \par
  \end{beamercolorbox}
  \vfill
  \end{frame}
}

  \vfill
  \centering

  \end{beamercolorbox}
  \vfill
  \end{frame}
}
